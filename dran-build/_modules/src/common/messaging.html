[docs]def msg_wrapper(log_name, log, msg):
    """ Wraps logging messages. 
    
        Args:
            log_name (str)  : The name of the logger e.g. info
            log (object)    : The logging object
            msg (str)       : The message to be wrapped.
    """

    # setup wrappers
    wrappers = {
        'error': "* ERROR * : ",
        'warning': "WARNING! : ",
        'info': "# >>> ",
        'debug': "> DEBUG : ",
        'basic': "# -- "
    }

    log("\n"+wrappers[log_name]+msg)
